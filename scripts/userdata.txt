#!/bin/bash

yum update -y
yum install -y ruby aws-cli

cd /home/ec2-user

wget https://aws-codedeploy-us-west-2.s3.amazonaws.com/latest/install
chmod +x ./install
sudo ./install auto

wget -qO- https://raw.githubusercontent.com/creationix/nvm/v0.33.6/install.sh | bash
. ~/.nvm/nvm.sh
nvm install 8.9.1
npm i -g pm2@latest
